Hello, Thakyou for using Panther. Here are the commands that Panther supports ---------


var	Declare a variable	        var x = 10;
const	Declare a constant	        const PI = 3.14;
If	Conditional statement	        If (x > 5) { ... }
Else	Alternative condition	        } Else { ... }
While	While loop	                While (x < 10) { ... }
For	For loop	                For (var i=0; i<5; i=i+1) { ... }
Loop	Simple counter loop	        Loop(5) { ... }
Print	Output to console	        Print("Hello");
Accept	Get user input	                Accept(name, "Name: ");
Run	Execute another .pnr file	Run("script.pnr");
--run	Activate engine (interactive)	--run
--exit	Exit interpreter	        --exit
--vars	Show all variables	        --vars
true	Boolean true	                var flag = true;
false	Boolean false	                var flag = false;
//	Single-line comment	        // This is a comment


Dns_Lookup()	Get IP addresses for domain	Dns_Lookup("google.com");
Reverse_IP()	Get hostnames for IP	        Reverse_IP("8.8.8.8");
Ping()	        Test connectivity	               Ping("8.8.8.8");
Scan_Port()	Check if port is open	        Scan_Port("8.8.8.8", 80);
HTTP_Headers()	Get HTTP response headers	HTTP_Headers("example.com");
URL_Status()	Check URL HTTP status	        URL_Status("https://example.com");
SSL_Info()	SSL certificate details	        SSL_Info("google.com");
Whois()	Domain registration info	        Whois("google.com");
IP_Info()	Geolocation and IP details	IP_Info("8.8.8.8");
Subdomain_Enum()  Find subdomains	        Subdomain_Enum("example.com");
Phone_Info()	Phone number intelligence	Phone_Info("+14155552671");
Recon()	        Full reconnaissance suite	        Recon("example.com");



WebFingerprint()	Detect server technology	WebFingerprint("example.com");
HTTPSecurityAudit()	Check security headers	        HTTPSecurityAudit("example.com");
PortProbe()	        Scan common ports	                PortProbe("8.8.8.8");
DNSIntel()	        Get DNS intelligence	                DNSIntel("google.com");
CheckHeadersSecurity()	Verify security headers	        CheckHeadersSecurity("example.com");
CheckOpenRedirect()	Test for open redirects	        CheckOpenRedirect("example.com");
CheckSSLvulnerabilities()	Check SSL/TLS issues	CheckSSLvulnerabilities("example.com");



PasswordLeakCheck()	Check if password in breaches	PasswordLeakCheck("password123");
HashCompute()	        Calculate MD5/SHA1/SHA256	        HashCompute("text", "md5");
Base64Encode()	        Encode to Base64	                Base64Encode("Panther");
Base64Decode()	        Decode from Base64	                Base64Decode("UGFudGhlcg==");
GeneratePassword()	Generate random password	GeneratePassword(16);
BruteForceSimulation()	Simulate brute force timing	BruteForceSimulation("admin", 1000000);


Traceroute()	Trace network route	                Traceroute("google.com");
GeoIP()	        Geographic IP location	                        GeoIP("8.8.8.8");
DomainAge()	Check domain age	                DomainAge("google.com");
EmailVerify()	Email syntax validation	                EmailVerify("user@example.com");
NetworkScan()	Simulate network scan	                NetworkScan("192.168.1.0/24", [80,443]);
Compute()	Display computation message             Compute("Scanning...");


Export_Report()	        Export to JSON or TXT	                Export_Report("json");
ExportHTMLReport()	Export to HTML	                ExportHTMLReport();
ClearReport()	        Clear report data	                ClearReport();
